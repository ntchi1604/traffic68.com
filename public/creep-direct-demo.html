CreepJS Direct Injection

Copy and paste this code into your website to inject CreepJS and get the visitorId:

<script src="/creep.js"></script>
<script src="/creep-injector.js"></script>
<script>
// Listen for the result
window.addEventListener('creep-result', function(e) {
 console.log('Creep result:', e.detail);
 var visitorId = e.detail.visitorId;
 var botData = e.detail.botDetection;

 // Use the visitorId here
 document.getElementById('visitor-id').textContent = visitorId;
 document.getElementById('is-bot').textContent = botData.bot ? 'Yes' : 'No';
});
</script>

<!-- Or use the API -->
<script>
// Wait for creep to load, then get visitorId
setTimeout(function() {
 var visitorId = window.CreepInjector.getVisitorId();
 console.log('Visitor ID:', visitorId);
}, 3000);
</script>
